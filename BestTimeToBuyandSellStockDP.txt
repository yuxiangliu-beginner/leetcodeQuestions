class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int minprice = INT_MAX;
        int maxProf = 0;
        for(int i = 0; i <prices.size();i++)
        {
            if(minprice>prices[i])
                minprice = prices[i];
            
            if(prices[i]-minprice > maxProf)
                maxProf = prices[i]-minprice;

                cout<<0<<endl;
        }
        return maxProf;
    }
};
